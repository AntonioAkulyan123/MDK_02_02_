CREATE TABLE [dbo].[Roles] (
	[Id] INT IDENTITY (1,1) NOT NULL,
	[NameRole] NVARCHAR (MAX) NULL,
	CONSTRAINT [PK_dbo.Roles] PRIMARY KEY CLUSTERED ([Id] ASC)
)

CREATE TABLE [dbo].[People] (
	[Id]	INT IDENTITY (1,1) NOT NULL,
	[RoleId]	INT  NOT NULL,
	[FirstName]	NVARCHAR (MAX) NULL, 
	[LastName] NVARCHAR (MAX) NOT NULL,
	[Birthday] DATETIME NOT NULL,
	CONSTRAINT [PK_dbo.People] PRIMARY KEY CLUSTERED ([Id] ASC),
	CONSTRAINT [FK_dbo.People_dbo.Roles_RoleId] FOREIGN KEY ([RoleId]) REFERENCES [dbo].[Roles] ([Id]) ON DELETE CASCADE
)

GO
CREATE NONCLUSTERED INDEX [IX_RoleId]
	ON [dbo].[People]([RoleId] ASC);
